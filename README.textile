# 'Server' application
require 'distribunaut'

configatron.distribunaut.app_name = :user_app
configatron.distribunaut.share_objects = true

class User
  include Distribunaut::Distributable

  attr_accessor :username

  def self.hi
    'hello!!!'
  end

  def save
    puts "Saving: #{self.inspect}"
  end

end

DRb.thread.join
